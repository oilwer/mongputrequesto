API documentation


API LOGIN

URI: http://localhost:3000/api/login

HTTP REQUEST: POST

DESC: Try to login

PAYLOAD: {"email": "email@domain.se", "password": "pw"}

RETURNS: bool


HTTP REQUEST: GET

DESC: Check if logged in

PAYLOAD: none

RETURNS: bool


HTTP REQUEST: DELETE

DESC: Logout

PAYLOAD: none

RETURNS: bool


API USER

URI: http://localhost:3000/api/user

HTTP REQUEST: POST

DESC: Adds user

PAYLOAD: {"email": "email@domain.se", "password": "pw"} (or all of the fields)

RETURNS: json


HTTP REQUEST: GET

URL: /api/user/557c4349779ef7d82cd8611a (DB ID)

DESC: Gets all the users

PAYLOAD: none

RETURNS: json


API COURSE

URI: http://localhost:3000/api/course

HTTP REQUEST: POST

DESC: Adds course

PAYLOAD: {"name":"course2", "code":"course2", "status":"true", "url":"course2", "teacher":"Berit", "description":"very nice course for you"}

RETURNS: json


HTTP REQUEST: GET

URL: /api/course/ (DB ID)

DESC: Gets all the courses

PAYLOAD: none

RETURNS: json


HTTP REQUEST: UPDATE


HTTP REQUEST: DELETE


HTTP REQUEST: UPDATE

URL: /api/user/557c4349779ef7d82cd8611a (DB ID)

DESC: Update user

PAYLOAD: {"email": "email@domain.se", "password": "pw"} (or all of the fields that you want to update)

RETURNS: json


HTTP REQUEST: DELETE

URL: /api/user/557c4349779ef7d82cd8611a (DB ID)

DESC: Delete user
